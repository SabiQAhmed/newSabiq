LogEntries logs = driver.manage().logs().get(LogType.BROWSER);
for (LogEntry entry : logs) {
    String message = entry.getMessage();
    if (message.contains("https://vbuildpguat")) {
        String[] split = message.split("https://");
        if (split.length > 1) {
            String fullUrl = "https://" + split[1].split("\"")[0];
            System.out.println("Captured URL: " + fullUrl);
            driver.navigate().to(fullUrl);  // Navigate to captured URL
            break;
        }
    }
}
