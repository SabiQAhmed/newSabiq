String consoleLog = "Console log with URL: https://... Not allowed to launch 'microsoft-edge:https://vbuildgp..." ; // replace with actual log string

Pattern pattern = Pattern.compile("Not allowed to launch 'microsoft-edge:(.*?)'");
Matcher matcher = pattern.matcher(consoleLog);

if (matcher.find()) {
    String edgeUrl = matcher.group(1);
    System.out.println("Extracted Edge URL: " + edgeUrl);
} else {
    System.out.println("No URL found");
}
